ğŸ” 2. Query Parameters (URL ke ? ke baad ka part)

ğŸ“Œ Example URL:
http://localhost:3000/search?name=raj&age=25

const http = require('http');
const url = require('url');

http.createServer((req, res) => {
  const parsed = url.parse(req.url, true);  // true to get query as object
  const query = parsed.query;

  res.write(`Hello ${query.name}, You are ${query.age} years old`);
  res.end();
}).listen(3000);

ğŸ§  4. Parsing Body (Most Important for POST/PUT)

ğŸŸ« Raw Node Example:

const http = require('http');

const server = http.createServer((req, res) => {
  if (req.method === 'POST') {
    let body = '';

    req.on('data', chunk => {
      body += chunk;
    });

    req.on('end', () => {
      console.log('Body:', body);
      res.end('Data received');
    });
  }
});

server.listen(3000);
